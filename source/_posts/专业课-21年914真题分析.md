```c
    int i=0,top;

    while(str[i]!='\0'){
        switch (str[i]){
            case '(': {
              Push(S,'(');
              break;
            }
            case '[': {
              Push(S,'[');
              break;
            }
            case '{': {
              Push(S,'{');
              break;
            }
            case ')': {
              Pop(S,e);
              if(e!='(')return false;
              break;
            }
            case ']':{
              Pop(S,e);
              if(e!='[')return false;
              break;
            }
            case '}':{
              Pop(S,e);
              if(e!='{')return false;
              break;
            }
            default: break;
        }
        i++;
    }
    if (isEmpty(S))return true;
    else return false;
}
```

```c
void SearchBST(BiTree &T,ElemType x){  // &为CPP的引用
	BiTree s,q,f;  
    q=T;
	while (q){
		if (q->data==x){
		q->count++;
		return ;
		}  //如果找到该值则计数加一
		f=q;
		if (q->data>x) q=q->lling;

    s=(BiTNode*)malloc(sizeof(BiTNode));   // 因为llink rlink 并不是BiTree类型，而是BiTNode

}
```

```c
			for(j=0;j<num;j++){// 列
				if(D[i][k]+D[k][j]<D[i][j]){	// ik到中间点 kj中间点转发
					D[i][j]=D[i][k]+D[k][j];
					Path[i][j]=Path[k][j];		
				}
			}
		}
	}
}
```
